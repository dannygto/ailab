          <Button
            variant="contained"
            onClick={handleNext}
            disabled={loading}
          >
            {loading ? (
              <CircularProgress size={24} color="inherit" />
            ) : activeStep === steps.length - 1 ? (
              '鍒涘缓瀹為獙'
            ) : (
              '涓嬩竴姝?
            )}
          </Button>
        </Box>
      </Paper>
    </Box>
  );
};

export default ExperimentCreate;
